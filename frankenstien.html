# Normalizing text into lowercase and lemmatizing words, removing stop words, punctuation and numbers for easy counting

def word_tokenization_normalization(text):

    text = text.lower() # lowercase
    doc = nlp(text)     # loading text into model

    words_normalized = []
    for word in doc:
        if word.text != '\n' \
        and not word.is_stop \
        and not word.is_punct \
        and not word.like_num \
        and len(word.text.strip()) > 2:
            word_lemmatized = str(word.lemma_)
            words_normalized.append(word_lemmatized)

    return words_normalized

# Returns a list of words in the text and how often they appear

def word_count(word_list):
    word_counts = {}
    for word in word_list:
      word = word.lower()
      if word in word_counts:
        word_counts[word] += 1
      else:
        word_counts[word] = 1
    return word_counts

# Prints a list of the top 15 most frequently used words in the text and how often they appear

def print_top_15_frequent_words(word_counts):
    sorted_word_counts = dict(sorted(word_counts.items(), key=operator.itemgetter(1), reverse=True))
    top_15_words = list(sorted_word_counts.items())[:15]  # Get the top 15 words and counts
    for word, count in top_15_words:
        print(f"{word}: {count}")

def print_top_10_frequent_adjectives(word_counts):

    adjectives = {word: count for word, count in word_counts.items() if nlp(word)[0].pos_ == "ADJ"}
    sorted_adjectives = dict(sorted(adjectives.items(), key=operator.itemgetter(1), reverse=True))
    top_10_adjectives = list(sorted_adjectives.items())[:10]
    for word, count in top_10_adjectives:
        print(f"{word}: {count}")

adjectives = []
